let dayNumber = prompt("Введіть число від 1 до 7:"); // Перетворюємо введене значення в число dayNumber = parseInt(dayNumber); // Визначаємо день неділі за введеним числом let dayName; switch (dayNumber) { case 1: dayName = "Неділя"; break; case 2: dayName = "Понеділок"; break; case 3: dayName = "Вівторок"; break; case 4: dayName = "Середа"; break; case 5: dayName = "Четвер"; break; case 6: dayName = "П'ятниця"; break; case 7: dayName = "Субота"; break; default: dayName = "Невідомий день"; } // Виводимо результат у вікні браузера document.write(`Для числа ${dayNumber} відповідний день неділі: ${dayName}.`);
